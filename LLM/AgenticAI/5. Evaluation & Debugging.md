
## Evaluation & Debugging for Front-Office AI Agents

Create frameworks to assess agent reasoning, troubleshoot complex behaviors, and ensure system reliability.

In high-stakes environments, **"it seemed to work"** isn't good enough. You need systematic frameworks to measure, trace, and validate agent behavior.

```python
from langgraph.graph import StateGraph, END
from langchain_openai import ChatOpenAI
from typing import TypedDict, List, Optional, Dict, Any
from datetime import datetime
from dataclasses import dataclass, field
from enum import Enum
import json
import time
import uuid

llm = ChatOpenAI(model="gpt-4o")

# ─────────────────────────────────────────────
# 1. TRACING — capture everything that happens
# ─────────────────────────────────────────────

class TraceEventType(Enum):
    NODE_ENTER    = "node_enter"
    NODE_EXIT     = "node_exit"
    LLM_CALL      = "llm_call"
    TOOL_CALL     = "tool_call"
    DECISION      = "decision"
    MEMORY_READ   = "memory_read"
    MEMORY_WRITE  = "memory_write"
    ESCALATION    = "escalation"
    ERROR         = "error"

@dataclass
class TraceEvent:
    event_id:   str
    run_id:     str
    event_type: TraceEventType
    node:       str
    timestamp:  str
    duration_ms: Optional[float]
    input:      Optional[Dict]
    output:     Optional[Dict]
    metadata:   Dict = field(default_factory=dict)

class AgentTracer:
    """Full execution tracer — captures every step."""

    def __init__(self):
        self.traces: Dict[str, List[TraceEvent]] = {}  # run_id → events

    def start_run(self) -> str:
        run_id = str(uuid.uuid4())
        self.traces[run_id] = []
        return run_id

    def log(self, run_id: str, event_type: TraceEventType,
            node: str, input: Dict = None, output: Dict = None,
            duration_ms: float = None, metadata: Dict = {}):

        event = TraceEvent(
            event_id    = str(uuid.uuid4()),
            run_id      = run_id,
            event_type  = event_type,
            node        = node,
            timestamp   = datetime.utcnow().isoformat(),
            duration_ms = duration_ms,
            input       = input,
            output      = output,
            metadata    = metadata
        )
        self.traces[run_id].append(event)
        return event

    def get_trace(self, run_id: str) -> List[TraceEvent]:
        return self.traces.get(run_id, [])

    def get_llm_calls(self, run_id: str) -> List[TraceEvent]:
        return [e for e in self.get_trace(run_id)
                if e.event_type == TraceEventType.LLM_CALL]

    def get_decisions(self, run_id: str) -> List[TraceEvent]:
        return [e for e in self.get_trace(run_id)
                if e.event_type == TraceEventType.DECISION]

    def print_trace(self, run_id: str):
        print(f"\n{'='*60}")
        print(f"TRACE: {run_id}")
        print(f"{'='*60}")
        for event in self.get_trace(run_id):
            duration = f"{event.duration_ms:.1f}ms" if event.duration_ms else ""
            print(f"  [{event.timestamp}] {event.event_type.value:<20} | {event.node:<25} | {duration}")
            if event.metadata:
                print(f"    → {event.metadata}")

tracer = AgentTracer()

# ─────────────────────────────────────────────
# 2. EVALUATION — score agent reasoning quality
# ─────────────────────────────────────────────

@dataclass
class EvalResult:
    run_id:              str
    task:                str
    score:               float         # 0.0 - 1.0
    reasoning_quality:   float
    factual_accuracy:    float
    decision_quality:    float
    compliance_check:    bool
    latency_ms:          float
    llm_calls:           int
    passed:              bool
    feedback:            str
    timestamp:           str = field(default_factory=lambda: datetime.utcnow().isoformat())

class AgentEvaluator:
    """LLM-as-judge + rule-based evaluation framework."""

    def __init__(self, tracer: AgentTracer):
        self.tracer = tracer
        self.results: List[EvalResult] = []

    # --- LLM-as-Judge ---
    def evaluate_reasoning(self, task: str, analysis: str, decision: str) -> Dict:
        prompt = f"""
        You are an expert evaluator for a front-office AI system.
        Score this agent's reasoning on each dimension from 0.0 to 1.0.

        TASK: {task}
        AGENT ANALYSIS: {analysis}
        AGENT DECISION: {decision}

        Evaluate on:
        1. reasoning_quality  - Is the logic sound and well-structured?
        2. factual_accuracy   - Are claims grounded and accurate?
        3. decision_quality   - Is the decision appropriate for the task?

        Respond ONLY in JSON:
        {{
            "reasoning_quality": 0.0-1.0,
            "factual_accuracy": 0.0-1.0,
            "decision_quality": 0.0-1.0,
            "feedback": "brief explanation"
        }}
        """
        result = llm.invoke(prompt)
        try:
            return json.loads(result.content)
        except:
            return {
                "reasoning_quality": 0.0,
                "factual_accuracy":  0.0,
                "decision_quality":  0.0,
                "feedback": "Evaluation parse failed"
            }

    # --- Rule-based compliance check ---
    def check_compliance(self, state: Dict) -> bool:
        rules = [
            state.get("audit_log") is not None,          # audit trail exists
            len(state.get("audit_log", [])) > 0,         # at least one log entry
            state.get("confidence") is not None,          # confidence was scored
            state.get("status") in ["complete",
                                    "escalated", "failed"] # valid terminal status
        ]
        return all(rules)

    # --- Full evaluation ---
    def evaluate_run(self, run_id: str, state: Dict, latency_ms: float) -> EvalResult:
        llm_calls = len(self.tracer.get_llm_calls(run_id))

        # LLM-as-judge scores
        scores = self.evaluate_reasoning(
            task     = state.get("raw_input", ""),
            analysis = state.get("current_analysis", ""),
            decision = state.get("decision", "")
        )

        # Rule-based compliance
        compliant = self.check_compliance(state)

        # Composite score
        composite = (
            scores["reasoning_quality"] * 0.35 +
            scores["factual_accuracy"]  * 0.35 +
            scores["decision_quality"]  * 0.30
        ) * (1.0 if compliant else 0.5)   # compliance failure halves score

        result = EvalResult(
            run_id            = run_id,
            task              = state.get("raw_input", ""),
            score             = round(composite, 3),
            reasoning_quality = scores["reasoning_quality"],
            factual_accuracy  = scores["factual_accuracy"],
            decision_quality  = scores["decision_quality"],
            compliance_check  = compliant,
            latency_ms        = latency_ms,
            llm_calls         = llm_calls,
            passed            = composite >= 0.75 and compliant,
            feedback          = scores["feedback"]
        )

        self.results.append(result)
        return result

    def print_eval(self, result: EvalResult):
        status = "✅ PASS" if result.passed else "❌ FAIL"
        print(f"\n{'='*60}")
        print(f"EVALUATION RESULT: {status}")
        print(f"{'='*60}")
        print(f"  Composite Score:    {result.score:.3f}")
        print(f"  Reasoning Quality:  {result.reasoning_quality:.3f}")
        print(f"  Factual Accuracy:   {result.factual_accuracy:.3f}")
        print(f"  Decision Quality:   {result.decision_quality:.3f}")
        print(f"  Compliance:         {'✅' if result.compliance_check else '❌'}")
        print(f"  Latency:            {result.latency_ms:.0f}ms")
        print(f"  LLM Calls:          {result.llm_calls}")
        print(f"  Feedback:           {result.feedback}")

evaluator = AgentEvaluator(tracer)

# 3. REGRESSION TESTING — golden test suite
# ─────────────────────────────────────────────

@dataclass
class TestCase:
    test_id:          str
    description:      str
    input:            Dict
    expected_status:  str
    expected_min_confidence: float
    should_escalate:  bool
    tags:             List[str] = field(default_factory=list)

class RegressionSuite:
    """Run a battery of test cases to catch regressions."""

    def __init__(self):
        self.test_cases: List[TestCase] = []
        self.results: List[Dict] = []

    def add_test(self, test: TestCase):
        self.test_cases.append(test)

    def run_test(self, test: TestCase, app) -> Dict:
        start = time.time()
        try:
            result = app.invoke(test.input)
            latency = (time.time() - start) * 1000

            # Assertions
            status_ok     = result["status"] == test.expected_status
            confidence_ok = result["confidence"] >= test.expected_min_confidence
            escalate_ok   = result["requires_escalation"] == test.should_escalate

            passed = status_ok and confidence_ok and escalate_ok

            return {
                "test_id":    test.test_id,
                "passed":     passed,
                "latency_ms": latency,
                "failures": {
                    "status":     None if status_ok     else f"expected={test.expected_status} got={result['status']}",
                    "confidence": None if confidence_ok else f"expected>={test.expected_min_confidence} got={result['confidence']}",
                    "escalation": None if escalate_ok   else f"expected={test.should_escalate} got={result['requires_escalation']}"
                }
            }
        except Exception as e:
            return {
                "test_id": test.test_id,
                "passed":  False,
                "error":   str(e)
            }

    def run_all(self, app) -> Dict:
        print(f"\n{'='*60}")
        print(f"REGRESSION SUITE: {len(self.test_cases)} tests")
        print(f"{'='*60}")

        passed = failed = 0
        for test in self.test_cases:
            result = self.run_test(test, app)
            self.results.append(result)

            status = "✅" if result["passed"] else "❌"
            print(f"  {status} [{test.test_id}] {test.description}")

            if not result["passed"] and "failures" in result:
                for k, v in result["failures"].items():
                    if v:
                        print(f"       ↳ {k}: {v}")

            if result["passed"]:
                passed += 1
            else:
                failed += 1

        print(f"\n  Results: {passed} passed / {failed} failed")
        return {"passed": passed, "failed": failed, "results": self.results}

# 4. INSTRUMENTED AGENT STATE & NODES
# ─────────────────────────────────────────────

class FrontOfficeState(TypedDict):
    run_id:              str
    task_id:             str
    task_type:           str
    raw_input:           str
    current_analysis:    str
    decision:            str
    confidence:          float
    requires_escalation: bool
    response:            str
    status:              str
    audit_log:           List[str]

def instrumented_node(node_fn):
    """Decorator — wraps any node with automatic tracing."""
    def wrapper(state: FrontOfficeState) -> FrontOfficeState:
        run_id = state.get("run_id", "unknown")
        node_name = node_fn.__name__
        start = time.time()

        tracer.log(run_id, TraceEventType.NODE_ENTER, node_name, input={"state_keys": list(state.keys())})

        result = node_fn(state)

        duration = (time.time() - start) * 1000
        tracer.log(run_id, TraceEventType.NODE_EXIT, node_name, output=result, duration_ms=duration)

        return result
    return wrapper

@instrumented_node
def analyze(state: FrontOfficeState) -> FrontOfficeState:
    start = time.time()

    prompt = f"""
    Analyze this front-office request and respond in JSON:
    {{
        "analysis": "your reasoning",
        "decision": "recommended action",
        "confidence": 0.0-1.0,
        "requires_escalation": true/false
    }}

    Task: {state['raw_input']}
    """

    result = llm.invoke(prompt)
    duration = (time.time() - start) * 1000

    # Trace the LLM call explicitly
    tracer.log(
        state["run_id"],
        TraceEventType.LLM_CALL,
        "analyze",
        input  = {"prompt_length": len(prompt)},
        output = {"response_length": len(result.content)},
        duration_ms = duration,
        metadata = {"model": "gpt-4o"}
    )

    try:
        parsed = json.loads(result.content)
    except:
        parsed = {
            "analysis": result.content,
            "decision": "manual review required",
            "confidence": 0.0,
            "requires_escalation": True
        }

    # Trace the decision
    tracer.log(
        state["run_id"],
        TraceEventType.DECISION,
        "analyze",
        metadata = {
            "confidence": parsed["confidence"],
            "requires_escalation": parsed["requires_escalation"]
        }
    )

    return {
        "current_analysis":    parsed["analysis"],
        "decision":            parsed["decision"],
        "confidence":          parsed["confidence"],
        "requires_escalation": parsed["requires_escalation"],
        "audit_log":           state["audit_log"] + [
            f"[{datetime.utcnow().isoformat()}] Analysis: confidence={parsed['confidence']}"
        ]
    }

@instrumented_node
def generate_response(state: FrontOfficeState) -> FrontOfficeState:
    result = llm.invoke(f"Generate a professional response for:\n{state['decision']}")
    return {
        "response": result.content,
        "status":   "complete",
        "audit_log": state["audit_log"] + [f"[{datetime.utcnow().isoformat()}] Response generated"]
    }

@instrumented_node
def escalate(state: FrontOfficeState) -> FrontOfficeState:
    tracer.log(state["run_id"], TraceEventType.ESCALATION, "escalate",
               metadata={"confidence": state["confidence"]})
    return {
        "response": f"Escalated. Task: {state['task_id']}",
        "status":   "escalated",
        "audit_log": state["audit_log"] + [f"[{datetime.utcnow().isoformat()}] ESCALATED"]
    }

def route_after_analysis(state: FrontOfficeState) -> str:
    return "escalate" if state["requires_escalation"] or state["confidence"] < 0.75 else "respond"

# ─────────────────────────────────────────────
# 5. BUILD & INSTRUMENT GRAPH
# ─────────────────────────────────────────────

graph = StateGraph(FrontOfficeState)
graph.add_node("analyze",           analyze)
graph.add_node("generate_response", generate_response)
graph.add_node("escalate",          escalate)
graph.set_entry_point("analyze")
graph.add_conditional_edges("analyze", route_after_analysis,
                             {"respond": "generate_response", "escalate": "escalate"})
graph.add_edge("generate_response", END)
graph.add_edge("escalate",          END)
app = graph.compile()

# ─────────────────────────────────────────────
# 6. INSTRUMENTED RUN WRAPPER
# ─────────────────────────────────────────────

def run_with_eval(task: str, task_type: str = "client_inquiry") -> Dict:
    run_id   = tracer.start_run()
    task_id  = str(uuid.uuid4())[:8]
    start    = time.time()

    initial_state = FrontOfficeState(
        run_id              = run_id,
        task_id             = task_id,
        task_type           = task_type,
        raw_input           = task,
        current_analysis    = "",
        decision            = "",
        confidence          = 0.0,
        requires_escalation = False,
        response            = "",
        status              = "",
        audit_log           = []
    )

    result   = app.invoke(initial_state)
    latency  = (time.time() - start) * 1000

    # Evaluate
    eval_result = evaluator.evaluate_run(run_id, result, latency)

    # Print outputs
    tracer.print_trace(run_id)
    evaluator.print_eval(eval_result)

    return {"state": result, "eval": eval_result}
```

## Agent Metrics Framework

---

### 1. Reliability

| Metric | Formula | Target |
|---|---|---|
| **Pass Rate** | successful_runs / total_runs | > 95% |
| **Escalation Rate** | escalations / total_runs | Baseline ± 10% |
| **Error Rate** | failed_runs / total_runs | < 1% |
| **Tool Failure Rate** | failed_tool_calls / total_tool_calls | < 2% |
| **Retry Rate** | runs_requiring_retry / total_runs | < 5% |
| **Hallucination Rate** | flagged_hallucinations / total_runs | < 1% |
| **Task Completion Rate** | completed_without_human / total_runs | > 90% |

---

### 2. Performance

| Metric | What to measure | Target |
|---|---|---|
| **End-to-End Latency** | p50, p95, p99 ms per run | p95 < 5s |
| **Node Latency** | per-node execution time | Identify bottlenecks |
| **LLM Calls per Run** | count of llm.invoke() calls | Minimize |
| **Token Usage** | input + output tokens per run | Track cost |
| **Throughput** | runs per minute | Capacity planning |
| **Time to First Token** | ms until streaming starts | < 500ms |
| **Queue Depth** | pending tasks at any time | < threshold |

### 3. Drift

Drift is the most **undertracked** category — models and data change silently.

| Metric | What it detects | How |
|---|---|---|
| **Confidence Score Drift** | Model becoming less/more certain over time | Rolling avg confidence vs baseline |
| **Escalation Rate Drift** | Sudden spike = model degradation | ± 2σ alert |
| **Decision Distribution Drift** | Shift in decision categories | KL divergence vs baseline |
| **Semantic Output Drift** | Response meaning changing | Embed outputs, track cosine similarity |
| **Latency Drift** | Gradual slowdown | p95 trend over 7/30 days |
| **Input Distribution Drift** | New task types appearing | Cluster inputs, flag outliers |
| **Tool Usage Drift** | Agent using different tools than expected | Compare tool call frequency |

```python
# Simple drift detector
from collections import deque
import numpy as np

class DriftDetector:
    def __init__(self, window=100, threshold=2.0):
        self.window    = deque(maxlen=window)
        self.baseline  = None
        self.threshold = threshold  # std deviations

    def record(self, value: float) -> dict:
        self.window.append(value)

        if len(self.window) < 20:
            return {"drift": False, "reason": "insufficient data"}

        if self.baseline is None:
            self.baseline = {"mean": np.mean(self.window),
                             "std":  np.std(self.window)}

        current_mean = np.mean(list(self.window)[-20:])
        z_score = abs(current_mean - self.baseline["mean"]) / (self.baseline["std"] + 1e-9)

        return {
            "drift":         z_score > self.threshold,
            "z_score":       round(z_score, 3),
            "current_mean":  round(current_mean, 3),
            "baseline_mean": round(self.baseline["mean"], 3)
        }

# Usage
confidence_drift = DriftDetector(window=100, threshold=2.0)
result = confidence_drift.record(0.61)
# → {"drift": True, "z_score": 2.4, ...} triggers alert
```

### 4. Quality

| Metric | What it measures | Method |
|---|---|---|
| **Reasoning Score** | Logic soundness | LLM-as-judge (0-1) |
| **Factual Accuracy** | Grounded claims | LLM-as-judge vs ground truth |
| **Decision Quality** | Appropriate action taken | LLM-as-judge |
| **Response Coherence** | Output is clear and relevant | LLM-as-judge |
| **Context Utilization** | Did agent use available memory/context? | Check if retrieved facts appear in reasoning |
| **Groundedness** | Response supported by retrieved docs | RAG faithfulness score |
| **Human Override Rate** | How often humans reverse agent decisions | Track post-escalation outcomes |

```python
# Quality scorecard
@dataclass
class QualityScore:
    run_id:            str
    reasoning:         float   # 0-1
    factual_accuracy:  float   # 0-1
    decision_quality:  float   # 0-1
    coherence:         float   # 0-1
    groundedness:      float   # 0-1

    @property
    def composite(self) -> float:
        weights = {
            "reasoning":        0.25,
            "factual_accuracy": 0.30,
            "decision_quality": 0.25,
            "coherence":        0.10,
            "groundedness":     0.10
        }
        return (
            self.reasoning        * weights["reasoning"]        +
            self.factual_accuracy * weights["factual_accuracy"] +
            self.decision_quality * weights["decision_quality"] +
            self.coherence        * weights["coherence"]        +
            self.groundedness     * weights["groundedness"]
        )

    @property
    def passed(self) -> bool:
        return self.composite >= 0.75
```

### 5. Compliance

Non-negotiable in front-office. Every run must be auditable.

| Metric | What it checks | Must be |
|---|---|---|
| **Audit Log Presence** | Every run has a log | 100% |
| **Audit Log Completeness** | All required fields present | 100% |
| **Escalation Compliance** | High-risk decisions were escalated | 100% |
| **Decision Latency SLA** | Time-sensitive decisions met SLA | 100% |
| **PII Handling** | No raw PII leaked in logs | 0 violations |
| **Model Version Pinning** | Known model used, not drifted | 100% |
| **Human Override Logging** | All overrides captured | 100% |
| **Retention Policy** | Logs retained per policy | 100% |

```python
class ComplianceChecker:

    REQUIRED_AUDIT_FIELDS = [
        "task_id", "run_id", "timestamp",
        "decision", "confidence", "status", "audit_log"
    ]

    def check(self, state: dict) -> dict:
        failures = []

        # 1. Required fields
        for field in self.REQUIRED_AUDIT_FIELDS:
            if not state.get(field):
                failures.append(f"Missing required field: {field}")

        # 2. Audit log not empty
        if not state.get("audit_log"):
            failures.append("Audit log is empty")

        # 3. High-risk decisions must be escalated
        if state.get("confidence", 1.0) < 0.75 and state.get("status") != "escalated":
            failures.append("Low-confidence decision not escalated")

        # 4. Terminal status is valid
        if state.get("status") not in ["complete", "escalated", "failed"]:
            failures.append(f"Invalid terminal status: {state.get('status')}")

        # 5. No PII in logs (basic check)
        log_text = " ".join(state.get("audit_log", []))
        pii_patterns = ["ssn", "passport", "date of birth", "dob"]
        for pattern in pii_patterns:
            if pattern in log_text.lower():
                failures.append(f"Potential PII in audit log: {pattern}")

        return {
            "compliant": len(failures) == 0,
            "failures":  failures,
            "score":     1.0 if not failures else round(1 - len(failures) * 0.2, 2)
        }
```

### Putting It All Together — Metrics Dashboard

```python
@dataclass
class AgentHealthReport:
    window_size:      int
    timestamp:        str

    # Reliability
    pass_rate:        float
    error_rate:       float
    escalation_rate:  float

    # Performance
    p50_latency_ms:   float
    p95_latency_ms:   float
    avg_llm_calls:    float

    # Drift
    confidence_drift: dict
    escalation_drift: dict

    # Quality
    avg_quality:      float
    avg_reasoning:    float
    avg_accuracy:     float

    # Compliance
    compliance_rate:  float
    pii_violations:   int

    @property
    def overall_health(self) -> str:
        if (self.pass_rate > 0.95 and
            self.compliance_rate == 1.0 and
            not self.confidence_drift.get("drift") and
            self.avg_quality > 0.75):
            return "🟢 HEALTHY"
        elif self.compliance_rate < 1.0 or self.pii_violations > 0:
            return "🔴 CRITICAL"
        else:
            return "🟡 DEGRADED"
```

| Severity | Condition | Action |
|---|---|---|
